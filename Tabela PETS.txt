CREATE DATABASE petshop;
USE petshop;

CREATE TABLE pets (
	codigo			 INT(11)	AUTO_INCREMENT PRIMARY KEY,
	nome			 VARCHAR(30),
	sexo			 CHAR(1),
	idade			 INT(2),
	tipo			 CHAR(1),
	peso			 FLOAT(7,2),
	vacinado		 BOOLEAN,
	codigoDono		 INT(11),
	ultimaVisita	 DATE,
	medico			 VARCHAR(30),
	obs			 	 TEXT,
	foto			 VARCHAR(100)
);
